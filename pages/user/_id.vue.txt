<template>
  <section class="container">
    <img src="~assets/img/logo.png" alt="Nuxt.js Logo" class="logo" />
    <h1 class="title">
      User
    </h1>
    <h2 class="info">
      {{ user.name }}
    </h2>
    <nuxt-link class="button" to="/user">
      Users
    </nuxt-link>
  </section>
</template>

<script>
import axios from '~plugins/axios'

export default {
  name: 'id',
  asyncData ({ params, error }) {
    return axios.get('/api/access/user/' + params.id)
    .then((res) => {
      return { user: res.data }
    })
    .catch((e) => {
      var data = null
      if (e.response) {
        data = {
          statusCode: e.response.status,
          message: e.response.data.message || e.response.statusText,
          detail: {
            Request: e.response.request._header.trim(),
            Response: Object.keys(e.response.headers).map((key) => {
              return key + ': ' + e.response.headers[key]
            }).join('\n')
          }
        }
        if (e.response.data.stack) {
          data.detail['Stack Trace'] = e.response.data.stack
        } else if (e.response.data) {
          data.detail.Data = e.response.data
        }
      }
      error(data)
    })
  },
  head () {
    return {
      title: `User: ${this.user.name}`
    }
  },
  transition: {
    name: 'test',
    mode: 'out-in'
  }

}
</script>
